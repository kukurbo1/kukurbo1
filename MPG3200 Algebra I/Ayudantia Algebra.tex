\subsection{Problema 1}
Demuestre que la intersección de grupos es grupo. ¿Será cierto para la unión?
\begin{proof}
    Sean $H,K \leqslant G$, y consideramos la intersección $H \cap K$. Como $H$ es subgrupo de $G$, tenemos que $e \in H$. De la misma forma $e \in K$, y entonces $e \in H \cap K$, y por lo tanto $H \cap K \neq \varnothing$.

    Sean $h,k \in H \cap K$. Tenemos que $h,k \in H$, por lo que $hk^{-1} \in H$. Similarmente $h,k \in K$, y entonces $hk^{-1} \in K$. Por lo tanto $hk^{-1} \in H \cap K$, y como $h,k$ son arbitrarios, tenemos que $H \cap K \leqslant G$.
\end{proof}

\subsection{Problema 2}
Decida si los siguientes grupos son finitamente generados:
\begin{enumerate}[(a)]
    \item $(\Z,+)$
    \item $(\Z^n,+)$
    \item $(\Q,+)$
    \item $(\mu_n,\c)$
    \item $(\Q^*,\c)$
\end{enumerate}
\begin{proof}
    [\dem{a}]
    Notemos que todo entero $n >0$ se escribe de la forma
    \[n = \underbrace{1+\dots+1}_{n \text{ veces}}.\]
    También, todo entero $n<0$ se escribe de la forma
    \[n = \underbrace{-1-\dots-1}_{n \text{ veces}}.\]
    Además, $0 = 1-1$, y entonces $\Z = \gen{1}$, por lo que es finitamente generado.
\end{proof}
\begin{proof}
    [\dem{b}]
    Consideramos la base canónica de $\Z^n$
    \[\{e_1,\dots,e_n\} = \{(1,0,\dots,0), \dots, (0,\dots,0,1)\}.\]
    Sea $v = (a_1,\dots,a_n) \in \Z^n$, entonces
    \[v = a_1 e_1 + \dots + a_n e_n.\]
    Como $a_i \in \Z = \gen{1}$, tenemos que $a_i e_i \in \gen{e_i}$, y entonces $v \in \gen{e_1,\dots,e_n}$. Luego $\Z = \gen{e_1,\dots,e_n}$, y entonces es finitamente generado.
\end{proof}
\begin{proof}
    [\dem{c}]
    Veamos que $(\Q,+)$ no es finitamente generado. Por contradicción, supongamos que
    $$\Q = \gen{\frac{p_1}{q_1}, \dots, \frac{p_n}{q_n}}$$
    con $p_i,q_i \in \Z$ y $q_i \neq 0$ para todo $i$.
    Sea $r \in \Q$, entonces existen $\alpha_1,\dots,\alpha_n \in \Z$ tales que
    \[r = \sum_{i=1}^n \alpha_i \frac{p_i}{q_i}.\]
    Sea $Q = q_1 \cdots q_n$, entonces notemos que
    \[Qr = Q \sum_{i=1}^n \frac{\alpha_i p_i}{q_i} = \sum_{i=1}^n \alpha_i p_i \frac{Q}{q_i}.\]
    Como $q_i \mid Q$, tenemos que $Q/q_i \in \Z$ para todo $i$, y entonces $Qr \in \Z$ para todo $r \in \Q$. Notemos que $1/(Q+1) \in \Q$, pero tenemos que $Q/(Q+1) \notin \Z$, contradicción. Por lo tanto $\Q$ no es finitamente generado.
\end{proof}

\item Considere ahora dos conjuntos $A,B$ infinitos ¿Qué se puede decir de $\abs{A \times B}$ respecto de $\abs A$ y $\abs B$?
\begin{proof}
    Supongamos que $\abs A \leq \abs B$. Fijamos $a \in A$, y consideremos la función
    \begin{align*}
        f \colon B &\to B \times A \\
        x &\mapsto (x,a).
    \end{align*}
    Sean $x,y \in B$ tales que $f(x) = f(y)$. Entonces $(x,a) = (y,a)$, y por lo tanto $x=y$. Entonces $f$ es inyectiva y luego $\abs B \leq \abs{B \times A}$.
    Como $\abs A \leq \abs B$, existe una función inyectiva $g \colon A \embbed B$. Luego definimos
    \begin{align*}
        h \colon B \times A &\to B \times B \\
        (x,y) &\mapsto (g(x),y).
    \end{align*}
    Sean $(x_1,y_1), (x_2,y_2) \in A \times B$ tales que $h(x_1,y_1) = h(x_2,y_2)$.
    \[\implies (g(x_1),y_1) = h(x_1,y_1) = h(x_2,y_2) = (g(x_2),y_2)\]
    \[\implies g(x_1) = g(x_2), \quad y_1 = y_2\]
    \[\implies x_1 = x_2, \quad y_1 = y_2\]
    \[\implies (x_1,y_1) = (x_2,y_2).\]
    Por lo tanto $h$ es inyectiva, y entonces $\abs{A \times B} = \abs{B \times B} = \abs B$.
\end{proof}
\end{enumerate}

    \begin{align*}
        \chi \colon \P(X) &\to 2^X \\
        A &\mapsto \chi_A(x) = \begin{cases}
            1, &x \in A \\
            0, &x \notin A
        \end{cases}
    \end{align*}
    Veamos que es biyectiva.
    \begin{itemize}
        \item \wea{Inyectiva: }
        Sean $A,B \in \P(X)$ tales que $\chi(A) = \chi(B)$. Entonces $\chi_A(x) = \chi_B(x)$ para todo $x \in X$. Notemos que
        \[x \in A
        \iff \chi_A(x) = 1
        \iff \chi_B(x) = 1
        \iff x \in B.\]
        Por lo tanto $A = B$. Entonces $\chi$ es inyectiva.
        \item \wea{Sobreyectiva: }
        Sea $f \in 2^X$, y consideremos $S = \{x \in X \colon f(x) = 1\}$. Notemos que $\chi_S(x) = f(x)$ para todo $x \in X$, por lo que $S$ es preimagen de $f$. Entonces $\chi$ es sobreyectiva.
    \end{itemize}
    Concluimos entonces que $\chi$ es una biyección entre $\P(X)$ y $2^X$.
\end{proof}

\begin{definition}
    Dada una acción $G \acts X$, definimos
    \[\Fix(g) = \{x \in X \colon g \c x = x\}.\]
\end{definition}

\begin{lemma}
    [Burnside]
    Sea $G$ un grupo finito tal que $G \acts X$.  Sea $X/G$ el conjunto de órbitas de la acción. Entonces
    \[\abs{X \setminus G} = \frac{1}{\abs{G}} \sum_{g \in G} \abs{\Fix(g)}\]
\end{lemma}
\begin{proof}
    \begin{align*}
        \sum_{g\in G} \abs{\Fix(g)}
        &= \sum_{g \in G} \abs{\{x \in X \colon g \c x = x \}} \\
        &= \sum_{g \in G} \abs{\{(g,x) \in G \times X \colon g \c x = x\}} \\
        &= \sum_{x \in X} \abs{\{g \in G \colon g \c x = x\}} \\
        &= \sum_{x \in X} \abs{\stab_G(x)}.
    \end{align*}
    Para cada $x \in X$, existen biyecciones entre la órbita $\orb_G(x)$ y el conjunto de clases laterales izquierdas $G/\orb_G(x)$. Luego
    \[\abs{\orb_G(x)} = [G \colon \orb_G(x)] = \frac{\abs G}{\abs{\orb_G(x)}}\]
    \[\implies \sum_{x\in X} \abs{\orb_G(x)}
    = \sum_{x \in X} \frac{\abs{G}}{\abs{\orb_G(x)}}
    = \abs{G} \sum_{x \in X} \frac{1}{\orb_G(x)}.\]
    Tenemos también que $X = \bigcupdot_{\Ocal \in X/G} \Ocal$
    \begin{align*}
        \implies \sum_{x \in X} \frac{1}{\orb_G(x)}
        &= \sum_{\Ocal \in X/G} \sum_{x \in \Ocal} \frac{1}{\abs{\orb_G(x)}} \\
        &= \sum_{\Ocal \in X/G} \sum_{x \in \Ocal} \frac{1}{\abs\Ocal} \\
        &= \sum_{\Ocal \in X/G} \frac{\abs\Ocal}{\abs\Ocal} \\
        &= \sum_{\Ocal \in X/G} 1 = \abs{X/G}
    \end{align*}
    \[\implies \abs{G} \sum_{x \in X} \frac{1}{\abs{\orb_G(x)}} = \abs{G} \c \abs{X/G}.\]
\end{proof}

\begin{definition}
    Decimos que un grupo $G$ es \wea{soluble} si existe una secuencia $G_0, \dots, G_n$ subgrupos de $G$ tales que
    \begin{itemize}
        \item $G_0 = G$, $G_n = \{1_G\}$.
        \item $G_i \normal G_{i-1}$, $\forall i$.
        \item $G_i/G_{i-1}$ es abeliano $\forall i$.
    \end{itemize}
    Una \wea{serie central} de $G$ es una secuencia de subgrupos
    \[G = H_0 \supseteq H_1 \supseteq \dots \supseteq H_k = \{1_G\}\]
    tal que $\forall i$, $H_i \normal G$ y $H_{i-1}/H_i \subseteq Z(G/H_i)$.

    $G$ es \wea{nilpotente} si existe una serie central para $G$.
\end{definition}

\begin{obs}
    Si escribimos $G^{(1)} = [G,G]$, $G^{(i+1)} = [G^{(i)},G]$ para $i \geq 1$, entonces tenemos que $G^{(i)} \normal G$ y $G^{(i)}/G^{(i+1)} \subseteq Z(G/G^{(i+1)})$ para todo $i$.

    Luego $G$ es nilpotente si y solo si $\exists m \in \Z$ tal que $G^{(m)} = \{1_G\}$.
\end{obs}

\subsection{Problema 1}
\textbf{Sea $G$ un grupo generado por un subconjunto $X$. Considere $g=x_1^{e_1} \cdots x_n^{e_n} \in X, \ e_i=\pm 1$ con $n\geq 2$ una palabra reducida. Si toda palabra reducida $g$ no es igual a $e_g$ entonces $G$ es libre.}
\begin{proof}
    Consideremos la función $f \colon X \to G$ dada por $x \mapsto x$. Por la propiedad fundamental de grupos libres, existe un homomorfismo $\vphi \colon F(X) \to G$ tal que $f(x) = \vphi(x)$ para todo $x \in X$.
    
    Notemos que $\vphi$ es sobreyectiva: Todo elemento $g \in G$ se escribe de la forma
    \[g = x_1^{e_1} \dots x_n^{e_n}, \quad x_i \in X, \quad e_i \in \{\pm 1\}.\]
    Luego
    \begin{align*}
        g &= x_1^{e_1} \dots x_n^{e_n} \\
        &= f(x_1)^{e_1} \dots f(x_n)^{e_n} \\
        &= \vphi(x_1)^{e_1} \dots \vphi(x_n)^{e_n} \\
        &= \vphi(x_1^{e_1} \dots x_n^{e_n})
    \end{align*}
    y entonces todo elemento tiene preimagen.
    Luego
    \[F(X) \simeq \frac{F(X)}{\ker\vphi} \simeq G\]
\end{proof}
