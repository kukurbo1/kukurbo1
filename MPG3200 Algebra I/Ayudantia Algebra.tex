\begin{enumerate}

\item Demuestre que la intersección de grupos es grupo. ¿Puede decir lo mismo de la unión?
\begin{proof}
    Sean $H,K \leqslant G$. Veamos que $H \cap K \leqslant G$.
    Notemos que $e \in H$ y $e \in K$, por lo que $e \in H \cap K$, y entonces $H \cap K \neq \varnothing$.

    Sean $g,h \in H \cap K$. Luego $g,h \in H$, y como es subgrupo, $gh^{-1} \in H$. También $g,h \in K$, por lo que $gh^{-1} \in K$. Luego $gh^{-1} \in H \cap K$.

    Para la unión, consideremos $2\Z, 3\Z \leqslant \Z$. Notemos que $2,3 \in 2\Z \cup 3\Z$, pero $2+3 = 5 \notin 2\Z \cup 3\Z$. Por lo tanto $2\Z \cup 3\Z$ no es grupo.
\end{proof}

\item Decida si los siguientes grupos son finitamente generados. Justifique en ambos casos.

\begin{enumerate}
\item $(\mathbb{Z},+),$
\begin{proof}
    Consideremos el conjunto $S = \{1\} \subseteq \Z$. Sea $n \in \Z$, tenemos 3 casos:
    \begin{itemize}
        \item Si $n>0$, entonces $n = \sum_{i=1}^n 1 \in \gen{1}$.
        \item Si $n<0$, entonces $n = \sum_{i=1}^n -1 \in \gen{1}$.
        \item Si $n=0$, entonces $0 = 1-1 \in \gen{1}$.
    \end{itemize}
    Por lo tanto $\Z = \gen{1}$, y entonces es finitamente generado.
\end{proof}

\item $(\mathbb{Z}^{n},+),$
\begin{proof}
    Consideremos $e_1, \dots, e_n \in \Z^n$ la base canónica de $\Z^n$. Luego para todo $v \in \Z^n$, existen $\a_1, \dots, \a_n \in \Z$ tales que
    \[v = \br{\sum_{i=1}^{\a_1} e_1, \dots, \sum_{i=1}^{a_n} e_n} \in \gen{e_1,\dots,e_n}.\]
    Luego $\Z^n$ es finitamente generado.
\end{proof}

\item $(\mathbb{Q}^*,\cdot)$
\item $(\mu_n=\{ z \in \mathbb{C} : z^n=1\}, \cdot)$

\begin{lemma}
    Sea $G$ un grupo finito de orden $p^n$ con $p$ primo. Entonces $Z(G) \neq \{e\}$.
\end{lemma}
\begin{proof}
    La ecuación de clases es
    \[p^3 = \abs G = \abs{Z(G)} + \sum_i [G \colon C_G(x_i)]\]
    donde $Z(G)$ es el centro y $[G \colon C_G(x_i)]$ es el índice del centralizador, el cual es igual al tamaño de la clase de conjugación de $x_i$.

    Por Lagrange, $\abs{C_G(x_i)} \mid p^3$, por lo que $\abs{C_G(x_i)} \in \{p,p^2,p^3\}$. Para $x_i \notin Z(G)$, tenemos que $C_G(x_i) \neq G$, por lo que $\abs{C_G(x_i)} < p^3$. Luego
    \[[G \colon C_G(x_i)] = \abs G / \abs{C_G(x_i)} \in \set{\frac{p^3}{p}, \frac{p^3}{p^2}} = \{p,p^2\}.\]
    Luego $p$ divide a $[G \colon C_G(x_i)]$ para todo $i$, y entonces $[G \colon C_G(x_i)] \equiv 0 \pmod p$. Por la ecuación de clases, obtenemos que
    \[\abs{Z(G)} \equiv 0 \pmod p\]
    y entonces $\abs{Z(G)} \in \{p, p^2, p^3\}$. Por lo tanto $\abs{Z(G)} \neq 1$, y entonces $Z(G) \neq \{e\}$.
\end{proof}

  \[ x_1 \cdots x_n \in P. \]
  Como $P$ es un ideal primo, entonces existe $k \in \{1,\dots,n\}$ tal que $x_k \in P$. Contradicción, pues $x_k \in I_k \setminus P$.

  Por lo tanto existe $k \in \{1,\dots,n\}$ tal que $I_k \subset P$.
\end{proof}

\item Sea $A$ un anillo y sea $I$ un ideal en $A$. $I$ es un ideal primo si y solo si $A/I$ es un cuerpo.

\item Un anillo local, es decir que posee un solo ideal maximal no tiene elementos idempotentes distintos de $0$ o $1$. Un elemento $a$ de un anillo $A$ se dice idempotente si $a^2=a$.


\item Sea $A$ un anillo no trivial y sea $x$ un elemento nilpotente, es decir existe $n \in \mathbb{N}$ tal que $x^n =0$. Entonces $1+x$ es nilpotente. 

\item El nilradical de un anillo, definido como el conjunto de todos los elementos nilpotentes, coincide con la intersección de todos sus ideales primos.

\item Sea $A$ un anillo y sea $A[[x]]$ el anillo de series formales $f=\sum_{n=0}^\infty a_nx^n$ con coeficientes en $A$. Un elemento $f$ es una unidad en $A[[x]]$ si y solo si $a_0$ es una unidad en A.
\end{enumerate}

\item Sea $A$ un anillo y $A[x]$ el anillo de polinomios con coeficientes en $A$. Sea $f=a_0+a_1x+ \cdots +a_nx^n \in A[x]$. Pruebe las siguientes afirmaciones.

\begin{enumerate}
    \item $f$ es una unidad en $A[x]$ si y solo si $a_0$ es una unidad en $A$ y $a_1, \cdots a_n$ son nilpotentes.
    \item $f$ es nilpotente si y solo si $a_0, \cdots a_n$ son nilpotentes.
    \item $f$ es un divisor de cero si y solo si existe $a \neq 0 \in A$ tal que $af =0$.
    \item El radical de Jacobson de $A[x]$, que corresponde al conjunto de intersección de todos los ideales maximales es igual al nilradical.
\end{enumerate}
\end{enumerate}
