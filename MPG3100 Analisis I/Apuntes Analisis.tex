\section{Topología}
\subsection{Topologías y espacios métricos}

\begin{definition}
  Dado un conjunto $X$, un conjunto
  \[\tau \subseteq \P(X) = \{\O \colon \O \subseteq X\}\]
  se dice \wea{topología en $X$} si
  \begin{enumerate}[(1)]
    \item $\varnothing, X \in \tau$;
    \item Si $\O_1, \O_2 \in \tau \implies \O_1 \cap \O_2 \in \tau$;
    \item $\{\O_\a\}_{\a\in I} \subseteq \tau \implies \bigcup_{\a \in I} \O_\a \in \tau$.
  \end{enumerate}
\end{definition}

\begin{notation}\
  \begin{itemize}
    \item Si $\O \in \t$, $\O$ se llama \wea{abierto}.
    \item $(X,\tau)$ es un \wea{espacio topológico}.
    \item Si $a \in X$ y $\O \in \t$ tal que $a \in \O$, $\O$ se llama \wea{vecindad de $a$}.
  \end{itemize}
\end{notation}

\begin{example}\
  \begin{enumerate}[(1)]
    \item Para cualquier conjunto $X$ se tiene que $\t = \{\varnothing,X\}$ es una topología de $X$ (\wea{topología trivial}).
    \item $\t = \P(X)$ es la \wea{topología discreta}.
    \item Si $(X,d)$ es un espacio métrico, una topología es
    \[\t = \{\O \subseteq X \colon \forall a \in \O, \exists r > 0 \colon B_r(A) \subseteq \O\}.\]
    Esto incluye $\R^n$, espacios normados, espacios de Hilbert, etc.
  \end{enumerate}
\end{example}

\begin{proposition}
  $E \ubseteq X$ es abierto si y solo si $\forall a \in E$, $\exists \O_a$ vecindad de $a$ tal que $\O_a \subseteq E$.
\end{proposition}
\begin{proof}\
  \begin{itemize}
    \item \boxed\implies
    Si $a \in E$ y $E$ es abierto, entonces $E$ es vecindad de $a$ y $E \subseteq E$. \checkmark
    \item \boxed\impliedby
    Por demostrar: $E$ abierto.
    Si $a \in E$, por hipótesis $\exists \O_a$ tal que $a \in \O_a \subseteq E$, y por lo tanto
    \[\bigcup_{a \in E} \O_a \subseteq E \subseteq \bigcup_{a \in E} \O_a.\]
    Por lo tanto $E = \bigcup_{a \in E} \O_a$, y al ser unión de abiertos, tenemos que $E$ es abierto.
  \end{itemize}
\end{proof}

\begin{definition}
  La \wea{clausura} de un conjunto $E$ está dada por
  \[\bar{E} = \{a \in X \colon \forall \O_a \ni a, \ \O_a \cap E \neq \varnothing\}.\]
\end{definition}

\begin{definition}
  $E$ es cerrado si y solo si $\bar{E} = E$.
\end{definition}

\begin{proposition}
  $E$ es cerrado si y solo si $X \setminus E$ es abierto.
\end{proposition}
\begin{proof}
  \
  \begin{itemize}
    \item \boxed{\implies} Sea $a \in X \setminus E$. Si $\forall \O_a$ vecindad de $a$
    \begin{align*}
      \O_a \not\subseteq X \setminus E
      &\iff \exists b \in \O_a \cap E \\
      &\iff \O_a \cap E \neq \varnothing \\
      &\iff a \in \bar E = E.
    \end{align*}
    Pero esto contradice que $a \in X \setminus E$. Entonces existe $\O_a$ tal que $\O_a \subseteq X \setminus E$, por lo que $X \setminus E$ es abierto.

    \item \boxed{\impliedby} Por demostrar: $\bar E = E$.
    Si $a \in E$, entonces $\forall \O_a$ vecindad de $a$ se tiene que $$a \in \O_a \cap E$$ y entonces $E \subseteq \bar E$. Veamos ahora que $\bar E \subseteq E$.

    Sea $a \in \bar E$, entonces para toda vecindad $\O_a$ se tiene que
    \[\O_a \cap E \neq \varnothing. \tag{$\ast$}\]
    Supongamos que $a \in \bar E \setminus E$, entonces $a \in X \setminus E$. Como $X \setminus E$ es abierto, existe $\O_*$ vecindad de $a$ tal que $\O_* \subseteq X \setminus E$. Entonces $\O_* \cap E = \varnothing$, lo que contradice $(*)$.
    Concluimos que $\bar E \setminus E = \varnothing$.
  \end{itemize}
\end{proof}

\begin{definition}
  [Topología inducida]
  Si $(X,\t)$ es un espacio topológico e $Y \subseteq X$, la \wea{topología inducida en $Y$} es
  \[\t|_Y = \{\O \cap Y \colon \O \in \t\}.\]
\end{definition}

\begin{exercise}
  Verificar que $\t|_Y$ es topología.
\end{exercise}

\begin{obs}
  Con la topología inducida, los abiertos de $Y$ no son necesariamente abiertos en $X$.
\end{obs}

\begin{example}
  En $\R$ con la topología usual, si tomamos $Y = [0,2]$ entonces $[0,1)$ es abierto, ya que $[0,1) = [0,2] \cap (-1,1)$ y $(-1,1)$ es abierto en $\R$.
\end{example}

\begin{example}
  Consideremos la \wea{2-esfera}
  \[\S^2 = \{(x,y,z) \in \R^3 \colon x^+ y^2 + z^= 1\}.\]
  El conjunto
  \[\{(x,y,z) \in \S^2 \colon z>0\}\]
  es abierto en la topología inducida por $\R^3$.
\end{example}

\begin{definition}
  Sea $P \in X$. Decimos que $\B_p \subseteq \t$ es una \wea{base de la topología $\t$ en $p$} si
  \begin{enumerate}[(i)]
    \item Todos los elementos de $\B_p$ son vecindades de $p$;
    \item Para toda vecindad $U \ni p$, existe $B \in \B_p$ tal que $B \subseteq U$.
  \end{enumerate}
  Decimos que $\B$ es \wea{base de la topología} si $\forall p \in X$, $\exists \B_p \subseteq \B$ base de la topología en $p$.
\end{definition}

\begin{example}
  Si $(X,d)$ es un espacio métrico, entonces
  \[\{B_r(p) \colon r \in (0,\oo)\}\]
  es base en $p$.
  \[\{B_r(p) \colon r \in (0,\oo), p \in X\}\]
  es base de $\t$.
\end{example}

\begin{example}
  $X$ con la topología discreta,
  \[\{\{p\} \colon p \in X\}\]
  es base.
\end{example}

\begin{proposition}
  Sea $X \neq \varnothing$ y $\B \subseteq \P(X)$.
  $\B$ es base de alguna topología si y solo si
  \begin{enumerate}[(i)]
    \item $\B$ cubre $X$
    \item $B_1,B_2 \in \B $, $\forall p \in B_1 \cap B_2, \exists B_p \in \B$ tal que $p \in B_p \subseteq B_1 \cap B_2$.
  \end{enumerate}
\end{proposition}
\begin{proof}
  \
  \begin{itemize}
    \item \boxed{\implies}
    $(X,\t)$ es topología y $\B$ es base de $\t$.
    Como $\B$ es base para $p \in X$, entonces existe $B_p \in \B$ vecindad de $p$. Luego
    \[X \subseteq \bigcup_{p \in X} B_p \subseteq X\]
    por lo que $(i)$ se cumple, ya que
    \[X = \bigcup_{p \in X} B_p \subseteq \bigcup_{B \in \B} B \subseteq X.\]

    Veamos ahora (ii). Tomamos $B_1,B_2 \in \B$, entonces $B_1 \cap B_2$ es abierto:
    Si $B_1 \cap B_2 = \varnothing$, entonces ta está en la topología.
    Si $B_1 \cap B_2 \neq \varnothing$, entonces existe $p \in B_1 \cap B_2$. Por definición de base, existe $B$ vecindad de $p$ con $B \subseteq B_1 \cap B_2$.

    \item \boxed{\impliedby}
    Sea $\t = \{\varnothing\} \cup \{\bigcup_{\a \in I} B_\a \colon \{B_\a\}_{\a \in I} \subseteq \B\}$ el conjunto de uniones arbitrarias de elementos de $\B$. Veamos que $\t$ es topología y $\B$ es base de $\t$.
    \begin{itemize}
      \item $\varnothing \in \t$.
      \item Por (i), $X = \bigcup_{B \in \B} B \in \t$, y entonces $X \in \t$.
      \item Si tomamos $\O_\a = \bigcup_{\beta \in I_\a} B_\beta^\a \in \t$
      \[\bigcup_{\beta \in J} \O_\beta = \bigcup_{\a \in I} \bigcup_{\beta \in I_\a} B_\beta^\a \in \t.\]
      \item Sean $\O_1, \O_2 \in \t$, entonces
      \[\O_1 = \bigcup_{\a \in I_1} B_\a^1, \quad
      \O_2 = \bigcup_{\beta \in I_2} B_\b^2.\]
      Sea $p \in \O_1 \cap \O_2$, entonces $p \in \O_1$ y $p \in \O_2$.
      Luego existe $\a \in I_1$ tal que $p \in B_\a^1$ y existe $\b \in I_2$ tal que $p \in B_\b^2$. Por lo tanto $p \in B_\a^1 \cap B_\b^2$.
      Luego $\O_1 \cap \O_2 \in \t$.
    \end{itemize}
  \end{itemize}
\end{proof}

\begin{example}
  Si $(X,\t_X)$ e $(Y,\t_Y)$ son espacios topológicos, entonces
  \[\{\O_1 \times \O_2 \colon \O_1 \subseteq \t_X, \O_2 \subseteq \t_Y\}\]
  es una base para una topología en $X \times Y$.
\end{example}

\begin{definition}
  La topología dada por la base anterior se llama \wea{topología producto}.
\end{definition}

\begin{example}
  La topología producto en $\R \times \R$ es igual a la topología en $\R^2$ dada por la norma usual.
\end{example}

\begin{exercise}

\begin{definition}
  Sea $(X,\t)$ un espacio topológico. Si $C \subseteq X$, decimos que $\O$ es \wea{vecindad de $C$} si $\O\in\t$ y $C \subseteq \O$.
\end{definition}

\begin{definition}
  Sea $(X,\t)$ espacio topológico y $(a_n)_{n\in\N} \subseteq X$. Decimos que $\lim_{n\to\oo} a_n = a$ si $\forall \O$ vecindad de $a$, existe $n_0$ tal que $a_n \in \O$, $\forall n \geq n_0$.
\end{definition}

\begin{example}
  Si consideramos $X$ con la topología discreta, entonces una sucesión $(a_n)_{n\in\N}$ es convergente si y solo si $\exists n_0$ tal que $a_n = a$ para todo $n \geq n_0$. Es decir, las sucesiones que convergen son constantes desde un cierto punto en adelante.
\end{example}

\subsection{Axiomas de Separación}

\begin{definition}
  Sea $(X,\t)$ espacio topológico. Decimos que $X$ es:
  \begin{enumerate}[(1)]
    \item \wea{Tychonoff} si $\forall u,v \in X$, existen abiertos $\O_u, \O_v \in \t$ tales que
    \[u \in \O_u \setminus \O_v \quad\land\quad
    v \in \O_v \setminus \O_u.\]

    \item \wea{Hausdorff} si $\forall u,v \in X$, existen $\O_u, \O_v \in \t$ tales que
    \[u \in \O_u \quad\land\quad
    v \in \O_v \quad\land\quad
    \O_u \cap \O_v = \varnothing.\]

    \item \wea{Regular} si $X$ es Tychonoff y además para $F \subseteq X$ cerrado y $u \in X \setminus F$ existen $\O, \O_u \in \t$ tales que
    \[F \subseteq \O \quad\land\quad
    u \in \O_u \quad\land\quad
    \O \cap \O_u = \varnothing.\]

    \item \wea{Normal} si $X$ es Tychonoff y si $F_1, F_2 \subseteq X$ son cerrados entonces existen $\O_1,\O_2 \in \t$ tales que
    \[F_1 \subseteq \O_1 \quad\land\quad
    F_2 \subseteq \O_2 \quad\land\quad
    \O_1 \cap \O_2 = \varnothing.\]
  \end{enumerate}
\end{definition}

\begin{proposition}
  $(X,\t)$ es Tychonoff si y solo si $\{v\}$ es cerrado para cualquier $v \in X$.
\end{proposition}
\begin{proof}\
  \begin{itemize}
    \item \boxed{\implies} Supongamos que $(X,\t)$ es Tychonoff.
    Sea $v \in X$, y sea $u \in X \setminus \{v\}$. Luego existe $\O_u \in \t$ tal que $u \in \O_u$ y $v \notin \O_u$. Luego $u \in \O_u \subseteq X \setminus \{v\}$, y como $u$ es arbitrario, $X \setminus \{v\}$ es abierto. Luego $\{v\}$ es cerrado.

    \item \boxed{\impliedby} Supongamos que $\{v\}$ es cerrado para todo $v \in X$. Sean $u,v \in X$ tales que $u \neq v$. Entonces $u \in X \setminus \{v\}$. Como $X \setminus \{v\}$ es abierto, existe $\O_u$ vecindad de $u$ tal que $\O_u \subseteq X \setminus \{v\}$. Por lo tanto $v \notin \O_u$. De la misma forma existe $\O_v$ vecindad tal que $u \notin \O_v$ y $v \in \O_v$.
  \end{itemize}
\end{proof}

\begin{corollary}
  Normal $\subseteq$ Regular $\subseteq$ Hausdorff $\subseteq$ Tychonoff.
\end{corollary}

\begin{definition}
  Sea $(X,d)$ espacio métrico y sean $p \in X$, $C \subseteq X$. La \wea{distancia entre $p$ y $C$} es
  \[\dist(p,C) = \inf\{\dist(p,c) \colon c \in C\}.\]
\end{definition}

\begin{theorem}
  Si $(X,d)$ es un espacio métrico, entonces es normal.
\end{theorem}
\begin{proof}
  Tomemos $F_1,F_2$ cerrados y disjuntos. Definimos
  \begin{align*}
    \O_1 &= \{a \in X \colon \dist(a,F_1) < \dist(a,F_2)\}, \\
    \O_2 &= \{b \in X \colon \dist(b,F_2) < \dist(b,F_1)\}.
  \end{align*}
  Tenemos que $\O_1 \cap \O_2 = \varnothing$. Notemos que si $a \notin F_2$, entonces $\dist(a,F_2)>0$, pues si $\dist(a,F_2) = 0$ entonces existe $(a_n)_{n\in\N} \subseteq F_2$ tal que $\lim_{n\to\oo} a_n = a$, por lo que $a \in \bar{F_2} = F_2$, contradicción. Entonces $F_1 \subseteq \O_1$ y $F_2 \subseteq \O_2$.

  Veamos que $\O_i$ es abierto:
  Sea $a \in \O_1$, por demostrar que $\exists \e>0$ tal que si $d(a,b) < \e$ entonces $b \in \O_1$ (si $b \in B_\e(a)$ entonces $b \in \O$).
  Queremos ver
  \[d(b,F_1) \leq d(b,p_1), \quad \forall p_1 \in F_1.\]
  Notemos que $d(b,F_1) \leq d(a,b) + d(a,p_1)$. Luego $\forall d >0$, $\exists p_1 \in F_1$ tal que
  \[d(a,p_1) \leq \dist(a,F_1) + \delta.\]
  Luego $\forall \delta>0$,
  \[d(b,F_1) \leq d(a,b) + \dist(a,F_1) + \delta.\]
  Tomando $\delta \to 0$
  \begin{align*}
    \dist(b,F_1) &\leq d(a,b) + \dist(a,F_1) \\
    &\leq d(a,b) + \dist(a,F_1) - \dist(a,F_2) + \dist(a,F_2).
  \end{align*}
  De la misma manera:
  \[d(a,F_2) \leq d(a,b) + \dist(b,F_2)\]
  Luego
  \begin{align*}
    d(b,F_1) &\leq 2d(a,b) + \dist(a,F_1) - d(a,F_2) + \dist(b,F_2) \\
    \dist(b,F_1) - \dist(b,F_2) &\leq 2d(a,b) + \dist(a,F_1) - \dist(a,F_2).
  \end{align*}
  Tomamos $\e>0$ tal que $2\e < \dist(a,F_2) - \dist(a,F_1)$, y entonces $b \in \O$. Por lo tanto $(X,d)$ es normal.
\end{proof}

\begin{theorem}
  Sea $X$ Tychonoff. $X$ es normal si y solo si $\forall F \subseteq X$ cerrado y $U$ vecindad abierta de $F$, existe $\O$ abierto tal que
  \[F \subseteq \O \subseteq \bar\O \subseteq U.\]
\end{theorem}
\begin{proof}\
  \begin{itemize}
    \item \boxed{\implies} Sea $F$ cerrado y $U$ abierto con $F \subseteq U$. Consideremos $F_2 = X \setminus U$. Tenemos que $F_2$ es cerrado y $F \cap F_2 = \varnothing$. Luego como $X$ es normal, existen $\O_1,\O_2 \in \t$ tales que
    \[F \subseteq \O_1, \quad F_2 \subseteq \O_2, \quad \O_1 \cap \O_2 = \varnothing.\]
    Veamos que $\O_1$ cumple. Ta tenemos que $F \subseteq \O_1$, además $\O_1 \cap \O_2 = \varnothing$ y $\O_2 \supseteq X \setminus U$. Por lo tanto $X \setminus \O_2 \subseteq U$. Además $\O_1 \subseteq X \setminus \O_2$
    \[\implies \bar{\O_1} \subseteq \bar{X \setminus \O_2} = X \setminus \O_2 \subseteq U.\]
    Por lo tanto $F \subseteq \O_1 \subseteq \bar{\O_1} \subseteq U$.

    \item \boxed{\impliedby} Sean $F_1, F_2$ cerrados tales que $F_1 \cap F_2 = \varnothing$. Sea $U = X \setminus F_2$, entonces $F_1 \subseteq U$.
    Por propiedad, existe $\O \in \t$ tal que
    \[F_1 \subseteq \O \subseteq \bar\O \subseteq U = X \setminus F_2.\]
    Tomamos $\O_2 = X \setminus \bar\O \in \t$. Entonces $F_2 \subseteq \O_2$. Similar para $F_1$.
  \end{itemize}
\end{proof}

\begin{definition}
  Un espacio topológico $(X,\t)$ se dice \wea{primero contable} si $\forall p \in X$ existe una base de abiertos en $p$ tal que la base es numerable.
\end{definition}

\begin{definition}
  $(X,\t)$ se dice \wea{segundo contable} si existe una base de $\t$ numerable.
\end{definition}

\begin{example}
  Todo espacio métrico $(X,d)$ es primero contable. Todo punto $p \in X$ tiene base $\{B_r(p) \colon p \in \Q_{>0}\}$.
\end{example}

\begin{definition}
  Sea $(X,\t)$ espacio topológico. Decimos que $A \subseteq X$ es \wea{denso} si $\bar{A} = X$.
\end{definition}

\begin{definition}
  Un espacio topológico $(X,\t)$ se dice \wea{separable} si existe un subconjunto denso numerable.

\subsection{Funciones Continuas}

\begin{definition}
  Si $(X,d)$ e $(Y,\rho)$ son espacios métricos, un función $f \colon X \to Y$ se dice \wea{continua en $a \in X$} si $\forall \e>0, \exists \delta>0$ tal que $$d(a,b) < \delta \implies \rho(f(a),f(b)) < \e.$$
  La función se dice \wea{continua} si es continua en todo $a \in X$.
\end{definition}

\begin{obs}
  Esto es equivalente a que $\forall \e>0, \exists \delta>0$ tal que $$b \in B_\delta(a) \implies f(b) \in B_\e(f(a)).$$
\end{obs}

\begin{obs}
  Esto es equivalente a que si $\O$ es una vecindad de $f(a)$, entonces existe $U$ vecindad de $a$ tal que $f(U) \subseteq \O$.
\end{obs}

\begin{obs}
  Hay dos definiciones equivalentes:
  \begin{enumerate}[(1)]
    \item Si $\O$ es abierto en $Y$, entonces $f^{-1}(\O)$ es abierto en $X$.
    \item $f$ es continua en $a$ si $\forall a_n \to a \implies f(a_n) \to f(a)$.
  \end{enumerate}
\end{obs}

\begin{definition}
  Sean $(X,\t_X)$ e $(Y,\t_Y)$ espacios topológicos. Decimos que $f \colon X \to Y$ es \wea{continua en $a \in X$} si $\forall \O \in \t_Y$ vecindad de $f(a)$, existe $U \in \t_X$ vecindad de $a$ tal que $f(U) \subseteq \O$.
\end{definition}

\begin{proposition}
  Sean $(X,\t_X), (Y,\t_Y)$ espacios topológicos. Una función $f \colon X \to Y$ es continua si y solo si $\forall \O \in \t_Y$, $f^{-1}(\O) \in \t_X$.
\end{proposition}
\begin{proof}

    \item [\boxed{\impliedby}]
    Sea $a \in X$ y $\O_a$ vecindad de $f(a)$. Entonces $f^{-1}(\O_a) \in \t_X$. Además por construcción, $a \in f^{-1}(\O_a)$ y entonces $\exists U_a \in \t_X$ tal que $a \in U_a \subseteq f^{-1}(\O_a)$. Por lo tanto $f(U_a) \subseteq \O_a$.

\subsection{Compacidad}

\begin{definition}
  Sea $X$ un conjunto. Un \wea{cubrimiento} es una familia $\{\O_\a\}_\a$ tal que $\ds X = \bigcup_\a \O_\a$. Si $(X,\t)$ es un espacio topológico, decimos que $\{\O_\a\}_\a$ es un \wea{cubrimiento abierto} ( o \wea{cubrimiento por abiertos}) si los $\O_\a$ son abiertos.
\end{definition}

\begin{definition}
  Un espacio métrico $(X,d)$ se dice \wea{totalmente acotado} si $\forall \e>0, \exists \{x_i\}_{i=1}^n$ tal que $$X = \bigcup_{i=1}^n B_\e(x_i).$$
\end{definition}

\begin{definition}
  Un espacio métrico $(X,d)$ se dice \wea{compacto} si Para cualquier cubrimiento abierto de $X$ existe un subcubrimiento finito.
\end{definition}

\begin{theorem}
  En espacios métricos son equivalentes:
  \begin{enumerate}[(1)]
    \item Todo cubrimiento abierto tiene subcubrimiento finito;
    \item Toda sucesión tiene subsucesión convergente.
    \item Completo y totalmente acotado;
  \end{enumerate}
\end{theorem}

Todas estas propiedades sirven como definición de espacio métrico compacto.

\begin{definition}
  Un espacio topológico $(X,\t)$ se dice \wea{conexo} si
  \[\forall \O_1, \O_2 \in \t \cor{
    (\O_1 \cap \O_2 \ \land\ \O_1 \cup \O_2 = X) \implies \O_1, \O_2 \in \{\varnothing, X\}

  Si $E \subseteq X$, decimos que $E$ es \wea{conexo} si es conexo con la topología inducida.
\end{definition}

\begin{theorem}
  Si $X$ es conexo y $f \colon X \to Y$ es continua, entonces $f(X)$ es conexo.
\end{theorem}

\begin{exercise}
  En $\R$ los conexos son intervalos.
\end{exercise}

\begin{corollary}
  Si $X$ es conexo y $f \colon X \to \R$ continua, entonces $f(X)$ es un intervalo.
\end{corollary}

\begin{definition}
  $X$ es un espacio topológico tal que $\forall f \colon X \to \R$, $f(X)$ es intervalo, se dice que $X$ cumple la \wea{propiedad del valor intermedio} (abreviado como \wea{PVI}).
\end{definition}

\begin{theorem}
  $X$ cumple PVI si y solo si $X$ es conexo.
\end{theorem}
\begin{proof}
  Supongamos que $X$ no es conexo. Veamos que no cumple PVI.
  Si no es conexo, entonces existen $\O_1,\O_2$ abiertos tales que
  \[\O_1 \neq \varnothing \neq \O_2 \quad\land\quad
  \O_1 \cap \O_2 = \varnothing \quad\land\quad
  X = \O_1 \cup \O_2.\]
  Consideremos la función
  \[f(x) = \begin{cases}
    1, &x \in \O_1 \\
    0, &x \in \O_2
  \end{cases}\]
  Entonces
  \[f^{-1}((a,b)) = \begin{cases}
    \O_1, &\text{si } 1 \in (a,b), 0 \notin (a,b), \\
    \O_2, &\text{si } 0 \in (a,b), 1 \notin (a,b), \\
    X, &\text{si } 0,1 \in (a,b), \\
    \varnothing, &\text{si } 0,1 \notin (a,b).
  \end{cases}\]
  $f$ continua, entonces $f(X)$ no es intervalo.
\end{proof}

  Como $\ds \sum_{n=1}^\oo f_n(y)$ es uniformemente convergente, existe $n_0$ tal que
  \[\abs{\sum_{n=n_0}^\oo f_n(x)} < \frac{\e}{3}\]
  \[\implies \abs{\sum_{n=1}^{n_0-1} f_n(y) - f_n(y_0)} \leq \frac{\e}.\]
  Como cada $f_n$ es continua, $\exists U_n$ tal que
  \[y\in U_n \implies \abs{f_n(y) - f_n(y_0)} < \frac{\e}.\]
  Tomando $U = U_1 \cap \dots \cap U_{n_0-1}$, tenemos que
  \[y \in U \implies \abs{\sum_{n=1}^{n_0-1} f_n(y) - f_n(y_0)} \leq \frac{\e}{3}.\]
\end{proof}

\begin{theorem}
  [Tietze]
  Si $X$ es normal, $F \subseteq X$ es cerrado y $f \colon F \to [a,b]$ continua, entonces existe $g \colon X \to [a,b]$ tal que $g\vert_F = f$.
\end{theorem}
\begin{proof}
  Suponemos que $[a,b] = [-1,1]$. La idea es construir funciones $g_n \colon X \to [-1,1]$ continuas tales que $\abs{g_n(x)} \leq (2/3)^n$ y para $x \in F$
  \[\abs{f(x) - \sum_{i=1}^n g_i(x)} \leq \br{\frac{2}{3}}^n.\]
  Entonces la extensión de $f$ es la función
  \[g(x) = \sum_{i=1}^\oo g_i(x).\]
  \begin{claim}
    $\forall K>0$ y $h \colon F \to \R$ tal que $\abs{h}\leq K$, $\exists g$ continua tal que $\abs g < 2K/3$ en $X$ y $\abs{h-g} < 2K/3$.
  \end{claim}
  \begin{obs}
    Si la afirmación se cumple:
    \begin{itemize}
      \item Para $n=1$, tomamos $h=f$ y $K=1$, obtenemos $g_1$.
      \item Para $n=2$, tomamos $h = f-g_1$ y $K = 2/3$, obtenemos $g_2$ con $\abs{g_2} < 1/3$.
      \item Para $n$ cualquiera, $h = f - g_1 - \dots - g_{n-1}$ y $K = (2/3)^{n-1}$. Luego existe $g_n$ tal que
      \[\abs{g_n} \leq \br{\frac{2}{3}}^n \quad\land\quad \abs{f - \sum_{i=1}^n g_i} < \br{\frac{2}{3}}^n.\]
    \end{itemize}
  \end{obs}
  Para demostrar la afirmación, tomamos
  \[A = \{x \in F \colon h(x) \leq -\frac{K}{3}\},\]
  \[B = \{x \in F \colon h(x) \geq \frac{K}{3}\}.\]
  Por el Lema de Urysohn, $\exists g \colon X \to [-K/3,K/3]$ continua tal que
  \[g(x) = \begin{cases}
    -K/3, &x \in A \\
    K/3, &x \in B.
  \end{cases}\]
  En $A$ tenemos que
  \[-K \leq h(x) \leq -\frac{K}{3},\]
  \[-\frac{2K}{3} \leq h(x) - g(x) \leq 0.\]
  En $B$ tenemos que
  \[\frac K3 \leq h(x) \leq K,\]
  \[0 \leq h(x) - g(x) \leq \frac{2K}{3}.\]
  Entonces este es el $g$ de la afirmación.
\end{proof}

\begin{theorem}
  [Teorema de Metrización de Urysohn]
  Sea $X$ segundo contable. Entonces $X$ es normal si y solo si $X$ es metrizable.
\end{theorem}
\begin{proof}
  Ya sabemos que metrizable implica normal, así que supongamos que $X$ es segundo contable y normal, y mostremos que es metrizable.

  Sea $\{U_n\}_{n\in\N}$ base de abiertos numerables. Sea
  \[A = \{(n,m) \in \N \times \N \colon \bar{U_n} \subseteq U_m\}.\]
  Por el Lema de Urysohn, $\forall (n,m) \in A, \exists f_{n,m} \colon X \to [0,1]$ tal que $\abs{f_{n,m}} < 1$ y
  \[f_{n,m}(x) = \begin{cases}
    0, &x \in \bar{U_n} \\
    1, &x \in X \setminus U_m.
  \end{cases}\]
  Tomemos
  \[d(x,y) = \sum_{(n,m) \in A} \frac{\abs{f_{n,m}(x) - f_{n,m}(y)}}{2^{n+m}}.\]
  \begin{claim}
    $d$ es métrica en $X$.
  \end{claim}
  Veamos que
  \begin{itemize}
    \item \boxed{d(x,y) = 0 \iff x=y} Notemos que si $x=y$ entonces
    \[d(x,x) = \sum_{(n,m) \in A} \frac{\abs{f_{n,m}(x) - f_{n,m}(x)}}{2^{n+m}} = \sum_{(n,m) \in A} \frac{0}{2^{n+m}} = 0.\]
    Si $x \neq y$, entonces $\{x\} \subseteq X \setminus \{y\} \eqcolon U$. Como $X$ es normal, entonces existe $\O$ tal que
    \[
    \{x\} \subseteq \O \subseteq \bar\O \subseteq U.\]
    Tomamos $U_n$ vecindad de $x$ en $\O$ y $U_m = U$ (???)
    \[(n,m) \in A \implies f_{n,m}(x) = 0 \ \land \ f_{n,m}(y) = 1.\]
    \[\implies d(x,y) > 0.\]
    \item \boxed{d(x,y) = d(y,x)} Trivial.
    \item \boxed{d(x,y) \leq d(x,z) + d(z,y)} Trivial
  \end{itemize}
  \begin{claim}
    La topología de $(X,d)$ coincide con la topología original. Es decir, $\id \colon (X,\t) \to (X,d)$ es homeomorfismo.
  \end{claim}
  Notemos que $d(\cdot, y)$ es continua en $(X,\t)$, y entonces $\id$ es continua. Veamos que es cerrada. Sea $F$ cerrado en $(X,\t)$, queremos ver que $\id(F) = F$ es cerrado en $(X,d)$.

  Tomemos $x \notin F$, entonces existe $\O$ abierto tal que
  \[F \subseteq \O \subseteq \bar\O \subseteq X \setminus \{x\}.\]
  Por demostrar que
  \[\exists \e>0 \cor{y \in F \implies d(x,y) > \e}\]
\end{proof}

\subsubsection{Teorema de Stone-Weierstrass}

\begin{definition}
  Sea $(X,\tau)$ un espacio topológico.
  $C(X) = C(X;R) = \{f \colon X \to \R \mid f \text{ continua}\}$.
\end{definition}

\begin{definition}
  $$d(f,g) = \sup_{u\in X} \abs{f(u) - g(u)}.$$
  Si $X$ es compacto, entonces $$d(f,g) = \max_{u \in X} \abs{f(u) - g(u)}.$$
  La topología inducida por esta métrica se llama \wea{topología uniforme}.
\end{definition}

\begin{obs}
  $C(X)$ es un $\R$-espacio vectorial.
\end{obs}

\begin{definition}
  Decimos que $A \subseteq C(X)$ es una \wea{álgebra de funciones} si es un subespacio vectorial y si $\forall f,g \in A$, $fg \in A$.
\end{definition}

\begin{definition}
  Se dice que $A \subseteq C(X)$ \wea{separa puntos} si $$\forall u,v \in X, \exists f \in A \colon f(u) \neq f(v).$$
\end{definition}

\begin{theorem}[Stone-Weierstrass]
  Si $X$ es compacto y Hausdorff, y $A$ es un álgebra que separa puntos y que contiene las constantes, entonces $A$ es denso en $C(X)$ con la topología uniforme.
\end{theorem}

\begin{obs}
  Si $X$ es normal, el lema de Urysohn garantiza que $C(X)$ separa puntos. En general, no sabemos si $C(X)$ separa puntos.
\end{obs}

\begin{lemma}
  Sea $X$ compacto Hausdorff y $A$ un álgebra de funciones que separa puntos y contiene las constantes.
  Entonces para todo $F$ cerrado y todo $x_0 \in X \setminus F$, existe $U$ vecindad de $x_0$ tal que
  \[U \cap F = \varnothing \quad\land\quad
  \forall \e > 0, \exists h \in A \cor{
    h\vert_U < \e \quad\land\quad
    h\vert_F > 1-\e \quad\land\quad
    0 \leq h \leq 1
  }\]
\end{lemma}

\begin{lemma}
  Sean $X$ compacto Hausdorff y $A$ un álgebra de funciones que separa puntos y contiene las constantes. Entonces
  \[\forall F_1,F_2 \in \tau^c, \forall \e>0 \cor{
    F_1 \cap F_2 = \varnothing \implies \exists h \in A \br{
      h\vert_{F_1} < \e \quad\land\quad h\vert_{F_2} > 1-\e \quad\land\quad 0 \leq h \leq 1.
    }
  }\]
\end{lemma}

\begin{proof}
  [\dem{Stone-Weierstrass}]
  \begin{claim}
    Basta considerar $f \in C(X)$ tales que $0 \leq f(x) \leq 1$.
  \end{claim}
  En efecto, para $f$ cualquiera, si $c = \max_{u \in X} \abs{f(u)}$ ($X$ es compacto), tenemos que $c \in A$.
  Si $f \in A$, entonces $f+c \geq 0$, $f+c \in A$, y entonces
  \[\frac{f+c}{\abs{f+c}} \in A, \quad 0 \leq \frac{f+c}{\abs{f+c}} \leq 1.\]
  Tomemos $n \in \N$, $f \in C(X)$ con $0 \leq f \leq 1$. Definimos
  \begin{align*}
    C_j &= \set{u \in X \colon f(u) \leq \frac{j-1}{n}}, \\
    B_j &= \set{u \in X \colon f(u) \geq \frac{j}{n}}.
  \end{align*}
  Por el lema anterior, existen $g_j \in A$ tal que $g_j < 1/n$ en $C_j$, $g_j > 1 - 1/n$ en $B_j$ y $0 \leq g_j \leq 1$.
  Tomando
  \[g(x) = \frac{1}{n} \sum_{j=1}^n g_j \in A\]
  Tendríamos
  \[\tag{$*$} \max_{x \in X} \abs{f(x) - g(x)} < \frac{3}{n}.\]
  Demostremos $(*)$. Tomamos $k \in \{1,\dots,n\}$, y supongamos que $0 \leq f(x) \leq k/n$.
  Me rindo con la demostración.
\end{proof}

\subsubsection{Teorema del Punto Fijo de Banach}

\begin{definition}
  Sea $(X,d)$ espacio métrico, y $T \colon X \to X$ una función continua. Decimos que $T$ es \wea{Lipschitz de constante $L$} si
  \[\forall x,y \in X \cor{d(T(x),T(y)) \leq L d(x,y)}\]
  Si $L < 1$, entonces $T$ se dice \wea{contracción}.
\end{definition}

\begin{definition}
  Si $T \colon X \to X$, $u \in X$, se dice \wea{punto fijo} si $T(u) = u$.
\end{definition}

\begin{theorem}
  Si $(X,d)$ es métrico completo y $T \colon X \to X$ es contracción, entonces $T$ tiene un único punto fijo.
\end{theorem}
\begin{definition}
  Veamos primero que es único. Supongamos que $u,v$ son puntos fijos:
  \[T(u) = u, \quad T(v) = v\]
  \[\implies d(u,v) = d(T(u),T(v)) \leq L d(u,v).\]
  Como $L<1$, tenemos que $d(u,v) = 0$, y entonces $u=v$.

  Veamos que existe. Tomamos $u \in X$, y escribimos
  \[u_n = \underbrace{T \circ T \circ \dots \circ T}_{n \ veces}(u) = T^n(u).\]
  Queremos demostrar que $\{u_n\}$ es sucesión de Cauchy, ya que como $(X,d)$ es completo, tenemos que $u_n$ converge, y luego tendremos que demostrar que el límite es el punto fijo.

  Notemos que
  \begin{align*}
    d(u_{n+1},u_n) &= d(T(u_n),T(u_{n-1})) \\
    &\leq L d(u_n, u_{n-1}) \\
    &\leq L^2 d(u_{n-1},u_{n-2}) \\
    &\leq \dots \leq L^n d(u_1,u_0)
    = L^n d(T(u),u).
  \end{align*}
  Sean $n>k$, entonces
  \begin{align*}
    d(u_n,u_k) &\leq d(u_n,u_{n-1}) + d(u_{n-1}, u_k) \\
    &\leq \sum_{j=k+1}^n d(u_j,u_{j-1}) \\
    &\leq \sum_{j=k+1}^n L^{j-1} d(u_0,u_1) \\
    &= d(u_0,u_1) \frac{L^k - L^n}{1-L} \\
    &\leq d(u_0,u_1) \frac{L^k}{1-L}.
  \end{align*}
  Luego $\forall \e>0, \exists n_0$ tal que si $k,n \geq n_0$, entonces
  \[\frac{L^k}{1-L} d(u_0,u_1) < \e\]
  \[\implies d(u_n,u_k) \leq \frac{L^k}{1-L} d(u_0,u_1) < \e\]
  y entonces $\{u_n\}$ es de Cauchy, y como $(X,d)$ es completo, entonces $\{u_n\}$ converge. Escribimos
  \[\bar u = \lim_{n\to\oo} u_n\]
  Como $u_{n+1} = T(u_n)$, obtenemos que
  \[\bar{u} = \lim_{n\to\oo} u_n = \lim_{n\to\oo} u_{n+1} = \lim_{n\to\oo} T(u_n) = T(\bar u).\]
\end{definition}

\subsubsection{Teorema de Categoría de Baire}

\begin{definition}
  Sea $(X,\t)$ espacio topológico y $\O \subseteq X$. Se dice que $x$ es \wea{interior a $\O$} si $\exists U_x$ vecindad abierta de $x$ tal que $U_x \subseteq \O$.
\end{definition}

\begin{definition}
  $x$ es \wea{exterior} a $F$ si $x$ es interios de $F^c$.
\end{definition}

\begin{definition}
  $x \in \partial A$ si $\forall U_x$ vecindad de $x$,
  \[U_x \cap A \neq \varnothing \quad\land\quad U_x \cap A^c \neq \varnothing.\]
\end{definition}

\begin{theorem}
  Sea $(X,d)$ un espacio métrico completo.
  \begin{enumerate}[(i)]
    \item Si $\{\O_n\}_{n\in\N}$ son abiertos densos, entonces $\ds \bigcap_{n\in\N} \O_n$ es denso.
    \item Si $\{F_n\}_{n\in\N}$ son cerrados con interior vacío, entonces $\ds \bigcup_{n\in\N} F_n$ tiene interior vacío.
  \end{enumerate}
\end{theorem}
\begin{proof}
  Primero notemos que $(i) \iff (ii)$, por lo que basta con demostrar $(i)$.

  Tomemos $u \in X$ y $B_{r_0}(u)$, queremos demostrar que $B_{r_0}(u) \cap \bigcap_{n\in\N} \O_n \neq \varnothing$.

  Como $\O_1$ es denso, tenemos que $B_{r_0}(u) \cap \O_1 \neq \varnothing$. Luego existe $u_1 \in B_{r_0}(u) \cap \O_1$. Como este conjunto es abierto, existe $\tilde{r_1}$ tal que
  \[B_{\tilde{r_1}}(u_1) \subseteq B_{r_0}(u) \cap \O_1.\]
  Tomamos $r_1 < \tilde{r_1}$ tal que $$\bar{B_{r_1}(u_1)} \subseteq B_{\tilde{r_1}}(u_1) \subseteq B_{r_0}(u) \cap \O_1.$$
  Como $\O_2$ es denso y abierto, existe $u_2$ tal que
  \[\bar{B_{r_2}(u_2)} \subseteq B_{r_1}(u_1) \cap \O_2.\]
  En general, existen $u_n,r_n$ tal que
  \[\bar{B_{r_n}(u_n)} \subseteq B_{r_{n-1}}(u_{n-1}) \cap \O_n \subseteq B_{r_{n-2}}(u_n) \cap \O_{n-1}.\]
  Además podemos pedir $r_n < 1/n$.
  \begin{claim}
    $\ds \bigcap_{n\in\N} \bar{B_{r_n}(u_n)} \subseteq \bigcap_{i\in\N} \O_i.$
  \end{claim}
  \begin{claim}
    $\ds \bigcap \bar{B_{r_n}(u_n)} \neq \varnothing$.
  \end{claim}
  \begin{obs}
    $n>m$, entonces $u_n \in B_{r_m}(u_m)$, luego
    \[d(u_m,u_n) < r_m < \frac1m\]
    Luego para todo $n,m \geq n_0$,
    \[d(u_m,u_n) < \frac{1}{n_0} < \e\]
    y entonces $\{u_m\}$ es de Cauchy. Luego existe el límite
    \[\bar{u} = \lim_{n\to\oo} u_n.\]
  \end{obs}
  Por construcción, $\forall m \geq m_0$, $u_m \in \bar{B_{r_{m_0}}(u_{m_0})}$, y entonces
  \[\bar u = \lim_{n\to\oo} u_n \in B_{r_{m_0}}, \quad \forall m_0\]
  \[\implies \bar u \in \bigcap_{n\in\N} B_{r_n}.\]
\end{proof}

\begin{corollary}
  Si $\ds \bigcup_{m\in\N} F_m$ tiene interior no vacío, entonces existe $m_0$ tal que $F_{m_0}$ tiene interior no vacío.
\end{corollary}

\begin{corollary}

  entonces existe $\O$ abierto y $M>0$ tal que
  \[\forall f \in \Fcal, \forall x \in X, \quad \abs{f(x)} \leq M.\]
\end{corollary}

\subsection{Álgebras y \texorpdfstring{$\sigma$}{sigma}-álgebras}

\begin{definition}
  Dado $X$ un espacio cualquiera, $\A \subseteq \P(X)$ se dice \wea{álgebra} si
  \begin{enumerate}[(i)]
    \item $X \in \A$,
    \item $C \in \A \implies C^c \in \A$
    \item $\{A_i\}_{i=1}^n \subseteq \A \implies \bigcup_{i=1}^n A_i \in \A$.
  \end{enumerate}
\end{definition}

\begin{obs}
  Como $X \in \A$, entonces $\varnothing = X^c \in \A$.
  Además, (iii) y (ii) nos da que $\bigcap_{i=1}^n A_i \in \A$.
\end{obs}

\begin{definition}
  Si $X$ es un espacio cualquiera, $\A \subseteq \P(X)$ se dice \wea{$\sigma$-álgebra} si es un álgebra y además

\end{definition}

\begin{obs}
  Las topologías no son $\sigma$-álgebras en general.
\end{obs}

    \item $\{C \colon \abs{C}<\oo \lor \abs{C^c} < \oo\}$ es álgebra pero no es $\sigma$-álgebra.

\end{example}

\begin{definition}

  \begin{itemize}
    \item $\mu(\varnothing) = 0$,
    \item Si $\{A_i\}_{i\in\N} \subseteq \A$ son disjuntos de a pares, entonces $\mu\br{\bigcup_{i\in\N} A_i} = \sum_{i\in\N} \mu(A_i)$.
  \end{itemize}
  la función se dice de \wea{medida finitamente aditiva} si
  \begin{itemize}
    \item $\mu(\varnothing) = 0$,
    \item Si $\{A_i\}_{i=1}^n \subseteq \A$ son disjuntos de a pares, entonces $\mu\br{\bigcup_{i=1}^n A_i} = \sum_{i=1}^n \mu(A_i)$.
  \end{itemize}
  $(X,\A,\mu)$ se dice \wea{espacio de medida}.
\end{definition}

\begin{definition}

\begin{proposition}

\end{proposition}

\begin{proposition}

\end{proof}

\begin{proof}
  Si $A = \varnothing$, entonces $\varnothing \subseteq [0,1/n]$ para todo $n \in \N$, luego
  \[\lam^*(\varnothing) \leq \lam^*([0,1/n]) = 1/n, \quad \forall n \in \N.\]
  \[\implies \lam^*(\varnothing) = 0.\]
  Si $A \subseteq B$ y $B = \bigcup_{i \in \N} [a_i,b_i]$, entonces $A \subseteq \bigcup_{i \in \N} [a_i,b_i]$. Luego $\lam^*(A) \leq \lam^*(B)$.

  So $\{A_i\}$ es una familia de conjuntos tal que $\sum_{i\in\N} \lam^*(A_i) \neq \oo$ (si fuera $\oo$ estamos listos), entonces existe $\{[a_k,b_k]\}_{k \in \N}$ tal que
  \[\lam^*(A_i) \geq \sum_k (b_k-a_k) - \frac{\e}{2^i}\]
  y $A_i \subseteq \bigcup_{k \in \N} [a_k^i, b_k^i]$.
  Luego
  \begin{align*}
    \bigcup_{i \in \N} A_i
    &\subseteq \bigcup_{i \in \N} \bigcup_{k \in \N} [a_k^i, b_k^i] \\
    \lam^*\br{\bigcup_{i \in \N} A_i}
    &\leq \sum_{i \in \N} \br{\sum_{k \in \N} (b_k^i - a_k^i)} \\
    &\leq \sum_{i \in \N} \br{\lam^*(A_i) + \frac{\e}{2^i}} \\
    &\leq \sum_{i\in\N} \lam^*(A_i) + \e.
  \end{align*}
  Como es $\forall \e>0$, tenemos el resultado.
\end{proof}

\begin{theorem}
  Si $A$ es $\lam$-medible en $\R^n$, entonces
  \begin{enumerate}[(1)]
    \item $\lam(A) = \inf\{\lam(U) \colon U \in \t \land A \subseteq U\}$.
    \item $\lam(A) = \sup\{\lam(K) \colon K \subseteq A \text{ compacto}\}$.
  \end{enumerate}
\end{theorem}
\begin{proof}
  Si $A$ es medible, entonces existen $R_i = [a_1^i, b_1^i] \times \dots \times [a_n^i, b_n^i]$ con $i \in \N$ tales que
  \[A \subseteq \bigcup_{i \in \N} R_i.\]
  Luego
  \[\lam(A) + \e \geq \sum_{i\in\N} \abs{b_1^i - a_1^i} \cdot \dots \cdot \abs{b_n^i - a_n^i}\]
  Luego tomemos
  \[\tilde R_i = (a_1^i - 1/k, b_1^i + 1/k) \times \dots \times (a_n^i - 1/k, b_n^i + 1/k)\]
  Notemos que
  \[A \subseteq \bigcup_{i \in \N} R_i \subseteq \bigcup_{i \in \N} \tilde R_i \in \t\]
  Luego
  \begin{align*}
    \mu\br{\bigcup_{i \in \N} \tilde R_i}
    &\leq \sum \abs{b_1^i - a_1^i + \frac{2}{k}} \cdot \dots \cdot \abs{b_n^i - a_n^i + \frac{2}{k}} \\
    &\leq \sum \abs{b_1^i - a_1^i} \cdot \dots \cdot \abs{b_n^i - a_n^i}
    + \frac{2}{k} \sum (\dots)
    + \br{\frac{2}{k}}^2 \sum (\dots) + \dots + \br{\frac{2}{k}}^n.
  \end{align*}
  Si $\sum \abs{R_i} < \oo$, entonces
  \[\mu\br{\bigcup_{i \in \N} \tilde R_i} \leq \sum_i \abs{b_1^i - a_1^i} \cdot \dots \cdot \abs{b_n^i - a_n^i} + \frac{2}{k} C
  \leq \mu(A) + \e + \frac{2C}{k}.\]
  Luego $\lam\br{\bigcup \tilde R_i} \leq \lam(A) + \tilde \e$ para cualquier $\tilde\e>0$. Entonces
  \[\int\{\lam(U) \colon U \supseteq A\} \leq \lam(A).\]
  Como $\lam$ es mootona, tenemos que $\lam(A) \leq \lam(U)$ para todo $U \supseteq A$.
  Si $\abs{\sum \abs{R_i}}$ no es finita, entonces $\lam(A) = \oo$, y la igualdad se cumple automaticamente.

  Para (2) tenemos dos casos:
  \begin{itemize}
    \item Si $A$ es acotado, existe $C$ cerrado de medida finita tal que $A \subseteq C$. Por la parte anterior, existe $U \supseteq C \setminus A$ abierto, luego
    \[\lam(C \setminus A) + \e > \lam(U).\]
    Queremos $K = \C \setminus U$. Notemos que $K \subseteq A$.
    \begin{align*}
      \lam(C) &\leq \lam(K) + \lam(U) \\
      &\leq \lam(K) + \lam(C \setminus A) + \e \\
      &\leq \lam(K) + \lam(C) - \lam(A) + \e
    \end{align*}
    \[\lam(A) \leq \lam(K) + \e\]
    \[\implies \sup\{\lam(K) \colon A \supseteq K\} \geq \lam(A) \geq \lam(K), \quad \forall K \subseteq A.\]
    \item Si $A$ no es acotado, queremos mostrar que $\forall b$ tal que $\lam(A) > b$, $\exists K \subseteq A$ tal que $\lam(K) + \e \geq b$.
    Si $A_k = A \cap B_k(0)$ para $k \in \N$, afirmamos que
    \[\lam(A_k) < \oo \quad\land\quad \lim_{k\to\oo} \lam(A_k) = \lam(A).\]
    Basta demostrar esto, pues implicaría que $\exists k_0$ tal que $\lam(A_{k_0}) > b + \delta$ con $\delta$ pequeño, y por el caso anterior $\exists K \subseteq A_{k_0}$ tal que
    \[\frac\delta2 + \lam(K) > \lam(A_{k_0}).\]
    Veamos la afirmación:
    \[A = \bigcup_{k \in \N} A_k \implies \lam(A) = \lam\br{\bigcup_{k \in \N} A_k}.\]
    Tomemos $\ds B_k = A_k \setminus \bigcup_{i=1}^{k-1} A_i$, luego
    \[\bigcup_{k \in \N} B_k = \bigcup_{k \in \N} A_k.\]
    \begin{align*}
      \lam(A) &= \sum_{k \in \N} \lam(B_k) \\
      &= \lim_{N\to\oo} \sum_{k=1}^N \lam(B_k) \\
      &= \lim_{N\to\oo} \lam\br{\bigcup_{k=1}^N B_k} \\
      &= \lim_{N\to\oo} \lam(A_N).
    \end{align*}
  \end{itemize}
\end{proof}

\subsection{Funciones Medibles}

\begin{definition}
  Sean $(X,\A,\mu)$ e $(Y,\B,\nu)$ espacios de medida. Una función $f \colon X \to Y$ se dice \wea{medible} si $\forall B \in \B$, $f^{-1}(B) \in \A$.
\end{definition}

\begin{proposition}
  Si $\B$ es generada por $\mathcal{G}$ y $f^{-1}(G) \in \A$ para todo $G \in \mathcal{G}$, entonces $f$ es medible.
\end{proposition}
\begin{proof}
  Sea $M = \{B \subset Y \colon f^{-1}(B) \in \A\}$. Veamos que $M$ es \( \sigma \)-álgebra.
  \begin{itemize}
    \item $\varnothing \in M$, pues $f^{-1}(\varnothing) = \varnothing \in \A$.
    \item Si $B \in M$, entonces \( f^{-1}(B^c) = (f^{-1}(B))^c \), y como $f^{-1}(B) \in \A$, entonces $(f^{-1}(B))^c \in \A$. Luego $f^{-1}(B^c) \in \A \implies B^c \in M$.
    \item Si $\{B_i\}_i \subseteq M$, entonces 
    \[ f^{-1}\left( \bigcup_i B_i  \right) = \bigcup_i \underbrace{f^{-1}(B_i )}_{\in \A} \in \A \]
    Luego como $f^{-1}\left( \bigcup_i B_i \right) \in \A$, tenemos que $\bigcup_i B_i \in M$.
  \end{itemize}
  Por hipótesis, $\mathcal{G } \subseteq M$, y entonces $M$ contiene a la \( \sigma \)-álgebra generada por \(\mathcal{G}\).
\end{proof}

\begin{obs}
  Si $X,Y$ son espacios topológicos, entonces las funciones continuas son medibles. Pero las funciones medibles no son necesariamente continuas.

  En general, si $Y = \R^n$, hay 2 opciones posibles de \( \sigma \)-álgebras naturales: $M_{\lam^*}$ (medibles respecto a $\lam^*$) o los Borelianos.

  En general decimos que $f \colon X \to \R$ es \wea{medible} si es \wea{Borel-medible}. Si queremos considerar $M_{\lam^*}$ decimos \wea{Lebesgue-medible}.
\end{obs}

\end{proposition}

\begin{definition}
  Dada $\phi \colon X \to [0,\oo]$ simple con representación canónica 
  \[ \phi = \sum_{i=1}^N c_i \chi_{E_i} \quad\land\quad (i \neq j \implies E_i \cap E_j = \varnothing) \]
  Definimos 
  \[ \int_X \phi \ d\mu = \sum_{i=1}^N c_i \mu(E_i). \]
\end{definition}

\begin{obs}
  $c_i \geq 0$. En general podemos asumir $c_i > 0$.
\end{obs}

Para que la integral esté bien definida, necesitamos que sea independiente de la representación canónica elegida:
Si tenemos $\{E_i\}_{i=1}^N$ y $\{F_j\}_{j=1}^M$ tales que 
\[ \phi = \sum_{i=1}^N C_i \chi_{E_i} = \sum_{j=1}^M d_j \chi_{F_j} \]
podemos asumir $c_i, d_j > 0$ y además $\bigcup_{i=1}^N E_i = \bigcup_{j=1}^M F_j$.
Si $x \in E_i \cap F_j$, entonces $c_i = d_j$. Luego 
\[ \phi = \sum_{i=1}^N c_i \chi_{E_i}(x) = \sum_{i,j} c_i \chi_{E_i \cap F_j}(x) = \sum_{i,j} d_j \chi_{E_i \cap F_j}(x) \]

\begin{align*}
    \implies \int_X \phi \ d\mu
    &= \sum_{i=1}^N c_i \mu(E_i) \\
    &= \sum_{i=1}^N c_i \mu\br{\bigcup_{j=1}^M (E_i \cap F_j)} \\
    &= \sum_{i=1}^N c_i \sum_{j=1}^M \mu(E_i \cap F_j) \\
    &= \sum_{i,j} d_j \mu(E_i \cap F_j) \\
    &= \sum_{j=1}^M d_j \sum_{i=1}^N \mu(F_j \cap E_i) \\
    &= \sum_{j=1}^M d_j \mu\br{\bigcup_{i=1}^N (F_j \cap E_i)} \\
    &= \sum_{j=1}^M d_j \mu(F_j).
\end{align*}

\begin{proposition}
  Si $\ds \phi = \sum_{i=1}^N c_i \chi_{E_i} \geq 0$ y $E_i$ no son necesariamente disjuntos, entonces 
  \[ \int_X \phi \ d\mu = \sum_{i=1}^N c_i \mu(E_i). \]
  i.e.: La representación no tiene que ser necesariamente canónica.
\end{proposition}

\begin{obs}
  Si $\mu(E_i) = \oo$ y $c_i = 0$ vamos a asumir en este contexto $c_i \mu(E_i) = 0$.
\end{obs}

\subsection{Integrales de funciones medibles}

\begin{definition}
  Sea $f \colon X \to [-\oo,\oo]$ medible. Definimos 
  \[ f_+(x) = \max\{f(x),0\}, \quad f_-(x) = -\min\{f(x),0\} \]
  La \wea{integral de $f$} es 
  \[ \int_X f \ d\mu \coloneqq \int_X f_+ \ \mathrm{d}\mu - \int_X f_- \ \mathrm{d}\mu \]
  Decimos que la integral existe si $\ds \int_X f_+$ o $\ds \int_X f_-$ es finita. Decimos que $f$ es \wea{integrable} si ambas son finitas.
\end{definition}

\begin{obs}
  $\abs f = f_+ + f_-$. Luego, si $f$ es integrable, entonces $\ds \int_X \abs f \ \mathrm{d}\mu$ existe.
\end{obs}

\begin{definition}
  El \wea{conjunto de funciones integrables} es 
  \[ \mathcal{L }_\mu^1 = \{f \colon X \to [-\oo,\oo] \text{ medible e integrable}\}. \]
\end{definition}

\begin{proposition}
  Sean $f,g \colon X \to \R$ medibles. 
  \begin{enumerate}[(1)]
    \item $\ds \int_X k f \ \mathrm{d}\mu = k \int_X f \ \mathrm{d}\mu$ para todo $k \in \R$.
    \item $\ds \int_X (f+g) \ \mathrm{d}\mu = \int_X f \ \mathrm{d}\mu + \int_X g \ \mathrm{d}\mu$.
    \item $\ds f \leq g \implies \int_X f \ \mathrm{d}\mu \leq \int_X g \ \mathrm{d}\mu$.
    \item $\ds \abs{\int_X f \ \mathrm{d}\mu} \leq \int_X \abs f \ \mathrm{d}\mu$.
  \end{enumerate}
\end{proposition}
\begin{proof}
  \ 
  \begin{enumerate}[(1)]
    \item Si $k \geq 0$ y $f = f_+ - f_-$, entonces 
    \[ (kf)_+ = kf_+, \quad (kf)_- = kf_- \]
    \begin{align*}
        \implies \int_X kf \ \mathrm{d}\mu
        &= \int_X (kf)_+ \ \mathrm{d}\mu - \int_X (kf)_- \ \mathrm{d}\mu \\
        &= \int_X kf_+ \ \mathrm{d}\mu - \int_X kf_- \ \mathrm{d}\mu \\
        &= k \int_X f_+ \ \mathrm{d}\mu - k \int_X f_- \ \mathrm{d}\mu \\
        &= k \br{\int_X f_+ \ \mathrm{d}\mu - \int_X f_- \ \mathrm{d}\mu} \\
        &= k \int_X f \ \mathrm{d}\mu.
    \end{align*}
    Si $k = -1$m entonces $kf = -f = f_- - f_+$. Se hace lo mismo.
    
    \item Si $f = f_+ - f_-$ y $g = g_+ - g_-$, entonces $f+g = f_+ + g_+ - (f_- + g_-)$, pero $f_+ + g_+ \neq (f+g)_+$. Lo que sí se cumple es 
    \[ (f_+ + g_+) - (f_- + g_-) = (f+g)_+ - (f+g)_- \]
    \[ \implies f_+ + g_+ + (f+g)_- = (f+g)_+ + (f_- + g_-). \]
    Como todas son positivas, tenemos 
    \begin{align*}
        \int f_+ + \int g_+ + \int (f+g)_-
        &= \int (f+g)_+ + \int f_- + \int g_- \\
        \implies \int f + \int g = \int (f+g).
    \end{align*}
    \item Usar que $g-f \geq 0$.
    \item $-\abs f \leq f \leq \abs f$.
  \end{enumerate}
\end{proof}

  [\wea{Fatou}]
  Si $f_n$ es una sucesión de funciones medibles no-negativas, entonces 
  \[ \liminf \int_X f_n \d\mu \geq \int_X \liminf_{n\to\oo} f_n \d\mu. \]
\end{lemma}

\begin{obs}
  Como soy lento para estas cosas, por definición de $\liminf$ queda 
  \[ \lim_{n\to\oo} \br{\inf_{m\geq n} \int_X f_m \d\mu} \geq \int_X \lim_{n\to\oo} \br{\inf_{m \geq n} f_m} \d\mu \]
  o también 
  \[ \sup_{n \geq 0} \inf_{m \geq n} \int_X f_m \d\mu \geq \int_X \sup_{n \geq 0} \inf_{m \geq n} f_m \d\mu. \]
\end{obs}

\begin{theorem}[\wea{Convergencia Dominada}]
  Sea $\{f_n\}$ una familia de funciones medibles tales que $\abs{f_n(x)} \leq g(x)$ a.e. y $g$ integrable. Si $\lim_{n\to\oo} f_n(x)$ existe (a.e.) y $f(x) = \lim_{n\to\oo} f_n(x)$, entonces $f$ es integrable.
\end{theorem}

\begin{obs}
  En convergencia monótona, lema de Fatou, la hipótesis de positividad se puede reemplazar por $f_n \geq 0$ con $g$ integrable.
\end{obs}

\subsection{Riemann Integrable vs Lebesgue Integrable}

\begin{theorem}
  Sea $f \colon [a,b] \to \R$.
  \begin{enumerate}[(a)]
    \item $f$ es Riemann integrable si y solo si es continua a.e. en $[a,b]$.
    \item Si $f$ es Riemann integrable, entonces es Lebesgue integrable, y su integral de Lebesgue es igual a su integral de Riemann.
  \end{enumerate}
\end{theorem}
\fecha{Clase 18. Miércoles 6 de Mayo}
\begin{proof}
  Escribimos $P = \{p_i\}_{i=0}^N$ con $a = p_0 \leq p_1 \leq \dots \leq p_N = b$.
  Para todo $i$ escribimos 
  \[ I_i = [p_i, p_{i+1}], \quad M_i = \sup_{x \in I_i} f(x), \quad m_i = \inf_{x \in I_i} f(x) \]
  Luego
  \begin{align*}
      U(f,P) &= \sum_{i=0}^{N-1} M_i \abs{p_{i+1} - p_i}, \\
      I(f,P) &= \sum_{i=0}^{N-1} m_i \abs{p_{i+1} - p_i}, \\
      \phi^*(f,P) &= \sum_{i=0}^{N-1} M_i \Chi{I_i}, \\
      \phi_*(f,P) &= \sum_{i=1}^{N-1} m_i \Chi{I_i}.
  \end{align*}
  $\phi^*$ y $\phi_*$ son funciones simples medibles. Sus integrales de Lebesgue son 
  \begin{align*}
      \int_{[a,b]} \phi^* \d\lam &= U(f,P), \\
      \int_{[a,b]} \phi_* \d\lam &= I(f,P).
  \end{align*}
  Además $\phi_* \leq f \leq \phi^*$.

  \begin{claim}
    $P \subseteq Q \implies U(f,P) \geq U(f,Q) \land I(f,P) \leq I(f,Q)$.
    En particular, para $P_1,P_2$ tenemos 
    \[ U(f,P_i) \geq U(f,P_1 \cup P_2) \quad\land\quad I(f,P_i) \leq I(f,P_1 \cup P_2). \]
  \end{claim}

  \begin{claim}
    Si $f$ es Riemann integrable, entonces existe $P_n$ tal que 
    \[ U(f,P_n) - I(f,P_n) < \frac{1}{n}. \]
  \end{claim}
  Por la primera afirmación, podemos asumir $P_{n+1} \subseteq P_n$. Luego 
  \[ \phi^*(f,P_n) \geq \phi^*(f,P_{n+1}), \quad \phi_*(f,P_n) \leq \phi_*(f,P_{n+1}). \]
  Como $f$ es acotada, existen constantes $c,k$ tales que 
  \[ c \leq \phi_* \leq \phi^* \leq k \]
  Tenemos también que $c,k$ son Lebesgue integrables en $[a,b]$. Escribimos 
  \[ \phi_n^* = \phi^*(f,P_n). \]
  La sucesión $(\phi_n^*)_n$ es decreciente y acotada, por lo tanto converge. En particular 
  \[ \lim_{n\to\oo} \phi_n^*(x) = \phi^*(x). \]
  Por convergencia dominada 
  \[ \lim_{n\to\oo} U(f,P_n) = \lim_{n\to\oo} \int_{[a,b]} \phi_n^* \d\lam = \int_{[a,b]} \phi^* \d\lam \]
  Similarmente $(\phi_n)_*(x) \xrightarrow{n\to\oo} \phi_*(x)$, y entonces
  \[ \lim_{n\to\oo} I(f,P_n) = \lim_{n\to\oo} \int_{[a,b]} (\phi_n)_* \d\lam = \int_{[a,b]} \phi_* \d\lam \]
  Luego
  \[ \int_{[a,b]} (\phi^* - \phi_*) \d\lam = 0. \]
  Además $(\phi_n)_* \leq (\phi_n)^*$, por lo que $\phi_* \leq \phi^*$. Por lo tanto $\phi^* - \phi_* = 0$ a.e.

  Tomemos $x$ tal que $\phi^*(x) = \phi_*(x)$, entonces $\forall \e>0$, $\exists n_0$ tal que para todo $n \geq n_0$:
  \[ \abs{\phi_n^*(x) - (\phi_n)_*(x)} < \e. \]
  Si $x \notin P_n$, entonces $x \in [p_i^n,p_{i+1}^n]$. Luego 
  \[ \abs{\sup_{I_i^n} f - \int_{I_i^n} f} < \e. \]
  Existe $\delta = \min\{p_{i+1}^n - x, x - p_i^n\}$ tal que
  \[ \abs{x-y} < \delta \implies \abs{f(x)-f(y)} \leq \abs{\sup_{I_i} f - \int_{I_i} f} < \e. \]
  Por otro lado, $\bigcup_{i \in \N} P_i$ es numerable, y por lo tanto es de medida 0. Entonces $f$ es continua en el complemento.

  Como $(\phi_n)_* \leq f \leq (\phi_n)^*$, entonces $f = \phi_* = \phi^*$ a.e.. Como $\phi_*$ es limite de medibles, es medible. Por lo tanto $f$ es medible. Luego 
  \[ \int (\phi_n)_* \d\lam \leq \int f \d\lam \leq \int_n^* \d\lam \]
  y tomando $n\to\oo$ obtenemos 
  \[ \int \phi_* \d\lam \leq \int f\d\lam \leq \int \phi^* \d\lam. \]
\end{proof}

\subsection{Convergencia de Funciones}

\begin{definition}
  Para todo $n \in \N$ sean $f,f_n \colon X \to \R$ medibles con $(X,\Acal,\mu)$ un espacio de medida.
  Decimos que \wea{$f_n \to f$ en medida} si $\forall \e>0$ 
  \[ \lim_{n\to\oo} \mu\{x \in X \colon \abs{f_n(x) - f(x)} > \e\} = 0 \]
\end{definition}

\begin{obs}
  \ 
  \begin{enumerate}[(1)]
    \item Se puede dar la misma definición con $f \colon X \to Y$ con $Y$ espacio métrico y \( \sigma \)-álgebra de Borel.
    \item Convergencia a.e. y en medida no son lo mismo. Por ejemplo $f_n = \Chi{[n,\oo)}$, tenemos que $f_n(x) \to 0$ a.e., pero si $\e<1$, entonces 
    \[ \lam\{x \colon f_n > \e\} = \oo. \]
    $f_n$ converge a.e., pero no en medida.
  \end{enumerate}
\end{obs}

\begin{proposition}
  Si $X$ es de medida finita, $\{f_n \colon X \to \R\}_n$ son medibles con $f_n \to f$ a.e., entonces $f_n \to f$ en medida.
\end{proposition}
\begin{proof}
  Queremos mostrar que $\forall \e>0$, 
  \[ \lim_{n\to\oo} \mu\{x \in X \colon \abs{f_n(x) - f(x)} > \e\} = 0. \]
  Tomemos 
  \[ A_n = \{x \in X \colon \abs{f_n(x) - f(x)} > \e\}, \quad B_n = \bigcup_{k \geq n} A_k. \]
  \[\implies \bigcap_{n\in\N} B_n = \bigcap_{n \in \N} \bigcup_{k \geq n} A_k = \limsup A_n.\]
  Por la definición 
  \[ B_n \supseteq B_{n+1} \]
  \[\implies \bigcap_{k=1}^n B_k = B_n.\]
  Queremos mostrar que $\lim_{n\to\oo} \mu(B_n) = \mu\br{\bigcap_{n\in\N} B_n}$.
  \begin{claim}
    \[ \bigcap_{n\in\N} B_n \subseteq \{x \in X \colon f_n(x) \not\to f(x)\}. \]
  \end{claim}
  Si $f_n \not\to f(x)$, entonces $\exists \e>0$ tal que $\forall n$, $\exists k \geq n$ tal que 
  \[ \abs{f_k(x) - f(x)} > \e_0. \]
  Luego $\mu\br{\bigcap_{n\in\N} B_n} = 0$, y entonces $\lim_{n\to\oo} \mu(B_n) = 0$. Notemos que, como $A_n \subseteq B_n$, entonces $\mu(A_n) \leq \mu(B_n)$, y entonces 
  \[ \lim_{n\to\oo} \mu(A_n) \leq \lim_{n\to\oo} \mu(B_n) = 0 \]
  \[ \implies \lim_{n\to\oo} \mu(A_n) = 0. \]
\end{proof}

\begin{proposition}
  Si $(X,\Acal,\mu)$ es un espacio de medida y $f_n,f$ son funciones medibles tales que $f_n \to f$ en medida, entonces existe una subsucesión $f_{n_k}$ tal que $f_{n_k} \to f$ a.e.
\end{proposition}
\begin{proof}
  Como $f_n \to f$ en medida, entonces $\forall k > 0, \exists n_k$ tal que 
  \[ \mu\underbrace{\set{x \in X \colon \abs{f_{n_k} - f(x)} > \frac{1}{k}}}_{\eqcolon A_k} \leq \frac{1}{2^k}. \]
  Además podemos elegir $n_{k+1} \geq n_k$.
  \begin{claim}
    Si $x \notin \bigcap_{n\in\N} \bigcup_{k \geq n} A_k = \limsup_{k\to\oo} A_k$, entonces $f_{n_k}(x) \to f(x)$.
  \end{claim}
  Si $x \notin \limsup A_k$, entonces existe $n \in \N$ tal que $\forall k \geq n$, $x \notin A_k$. Esto es si y solo si $\exists n \in \N$,$ \forall k\geq n$ 
  \[ \abs{f_{n_k}(x) - f(x)} < \frac{1}{k} \]
  Nos falta ver que 
  \[ \mu(\limsup A_k) = 0. \]
  Notemos que 
  \[ \mu(\limsup A_k) = \mu\br{\bigcap_{n \in \N} \bigcup_{k \geq n} A_n} \leq \mu\br{\bigcup_{k \geq n} A_k} \leq \sum_{k \geq n} \mu(A_k) \leq \sum_{k \geq n} \frac{1}{2^k} = \frac{1}{2^{n-1}}, \quad \forall n \in \N \]
  \[ \implies \mu(\limsup A_k) = 0. \]
\end{proof}

\begin{theorem}
  [\wea{Egoroff}]
  Sea $(X,\Acal,\mu)$ espacio de medida con $\mu$ medida finita $(\mu(X) < \oo$), y sea $f_n \to f$ a.e.. Entonces $\forall \e> 0,\exists B \subseteq X$ tal que $\mu(B) < \e$ y $f_n \to f$ uniformemente en $X \setminus B$.
\end{theorem}

\subsection{Teorema de Lusin}
Nuestro objetivo ahora es demostrar el siguiente teorema:
\begin{theorem}
  [\wea{Lusin}]
  Sea $X$ localmente compacto y Hausdorff con una \( \sigma \)-álgebra $\Acal$ que contiene a los borelianos. Supona que $\mu$ es regular en $A$ y $\sigma$-finita. Si $A \in \Acal$ es de medida finita, y $f \olon X \to \R$ es medible, entonces $\forall \e>0,\exists K \subseteq A$ compacto tal que $\mu(A \setminus K) < \e$ y $f|_K$ es continua. Además, $\exists g \colon X \to \R$ con soporte compacto y $f=g$ en $K$, $\sup_X \abs g \leq \sup_K \abs f$.
\end{theorem}

Antes de demostrar el teorema, necesitamos algunos preliminares.

\begin{definition}
  $(X,\t)$ es \wea{localmente compacto} si $\forall p \in X$, $\exists U \in \t$ tal que $p \in U$ y $\bar U$ es compacto.
\end{definition}

\begin{definition}
  Sea $X$ un espacio topológico Hausdorff y $(X,\Acal,\mu)$ espacio de medida tal que $\B(X) \subseteq \Acal$. $\mu$ se dice \wea{regular} si 
  \begin{enumerate}[(1)]
    \item $\forall K$ compacto, $\mu(K) < \oo$.
    \item $\forall A \in \Acal$, $\mu(A) = \inf\{\mu(U) \colon A \subseteq U \in \t\}$.
    \item $U \in \t \implies \mu(U) = \sup\{\mu(K) \colon K \subseteq U \text{ compacto}\}$.
  \end{enumerate}
\end{definition}

En el contexto anterior se puede tomar $X^* \supseteq X$ tal que $X^*$ es compacto y normal. Hoy asumiremos que $X$ es normal.

\begin{proposition}
  Si $X$ es Hausdorff y $\mu$ es $\sigma$-finita y regular, entonces $\forall A \in \Acal$, 
  \[ \mu(A) = \sup\{\mu(K) \colon K \subseteq A \text{ compacto}\}. \]
\end{proposition}

\begin{proposition}
  Si $X$ es normal y localmente compacto, $F$ compacto y $U$ es abierto, entonces existe $V$ abierto tal que 
  \[ F \subseteq V \subseteq \bar V \subseteq U \]
  con $\bar V$ compacto.
\end{proposition}

\newcommand{\simple}[4]{\sum_{#1 = 1}^{#2} {#3}_{#1} \Chi{{#4}_{#1}}}
\begin{proof}
  [\dem{Lusin}]
  Tomemos primero $f$ simple. 
  \[ f = \sum_{i=1}^N x_i \Chi{A_i} \]
  con $A_i$ disjuntos de a pares. Como $\mu$ es regular y $\mu(A) < \oo$, entonces existe $K_i \subseteq A_i \cap A$ tal que 
  \[ \mu(K_i) \geq \mu(A_i \cap A) + \frac\e2 \]
  Luego $K \coloneqq \bigcup_{i=1}^n K_i$ es compacto. 
  \[ f|_{K_1}(x) = \simple{i}{n}{c}{A_i \cap K_1}(x) = \simple{i}{n}{c}{K_i} \]
  Esta función es continua. Además, existe $\hat K$ compacto tal que $\hat K \subseteq A \setminus \bigcup_{i=1}^n A_i$. 
  \[ \frac{\e}{2^{n+1}} + \mu(\hat K) \geq \mu\br{A \setminus \bigcup_{i=1}^n A_i} \]
\end{proof}
